Groovy Script
//POST Create
def postmanPost = new URL('https://reqres.in/api/users')
def postConnection = postmanPost.openConnection()
postConnection.requestMethod = 'POST'
def form = "name: morpheus,job: leader"
postConnection.doOutput = true
def text
postConnection.with {
    outputStream.withWriter { outputStreamWriter ->
        outputStreamWriter << form
    }
    text = content.text
}
assert postConnection.responseCode == 201
JsonSlurper jsonSlurper = new JsonSlurper()
assert jsonSlurper.parseText(text)?.json.name == "morepheus"
